DROP TABLE STUDENT_DETAILS;
CREATE TABLE STUDENT_DETAILS (
    SLNO NUMBER PRIMARY KEY,
    
    COLUMN_NO NUMBER,

    F_V2PERSNO VARCHAR2(20) INVISIBLE,
    F_V2NAME VARCHAR2(100),
    F_V2QUALF VARCHAR2(50),
    F_V2BRANCH VARCHAR2(50),
    F_V2STREET VARCHAR2(150),
    F_V2DISTRICT VARCHAR2(100),
    F_V2STATE VARCHAR2(100),
    PINCODE VARCHAR2(10),
    F_V2UNIVERSITY VARCHAR2(150),
    F_V2GUIDE VARCHAR2(100),
    F_V2SECTION VARCHAR2(50),
    F_V2DIR VARCHAR2(100),
    PROJECTNAME VARCHAR2(200),
    COLLEGE VARCHAR2(150),
    DOCUMENTATION VARCHAR2(50),
    F_V2PHNO VARCHAR2(15),
    F_V2AADHAARNO VARCHAR2(20),
    PERFROM DATE,
    PERUPTO DATE
);
DELETE FROM STUDENT_DETAILS WHERE PERFROM IS NULL;
COMMIT;

CREATE OR REPLACE TRIGGER TRG_STUDENT_SLNO
BEFORE INSERT ON STUDENT_DETAILS
FOR EACH ROW
DECLARE
    v_yearMonth NUMBER;
    v_seq NUMBER;
BEGIN

    v_yearMonth := TO_NUMBER(TO_CHAR(:NEW.PERFROM,'YYYYMM'));

    SELECT NVL(MAX(MOD(SLNO,1000)),0)+1
    INTO v_seq
    FROM STUDENT_DETAILS
    WHERE FLOOR(SLNO/1000) = v_yearMonth;

    :NEW.SLNO := v_yearMonth*1000 + v_seq;

END;
/


TRUNCATE TABLE STUDENT_DETAILS;

DESC STUDENT_DETAILS;

ALTER TABLE STUDENT_DETAILS 
MODIFY DOCUMENTATION VARCHAR2(2000);
